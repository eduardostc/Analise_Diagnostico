const situacao = document.getElementById("id");

if(situacao){
    listarSituacao();
}

var empresas = [];
async function listarSituacao(){
    try {
        const dados = await fetch('listar_situacao.php');
    //console.log(dados)
    const resposta = await dados.json();
    // console.dir(resposta);

     empresas = resposta;

     //console.dir(empresas);

     if(resposta['status']){
        document.getElementById("msgAlertaID").innerHTML = "";
        for(var i = 0; i < resposta.dados.length; i++){
          // console.log(resposta.dados[i]['id']);
          //console.log(resposta.dados[i]['nome']);
            situacao.innerHTML = situacao.innerHTML + '<option value=" '+ resposta.dados[i]['id'] +' ">'
            + resposta.dados[i]['nome_da_empresa'] +'</option>'
        }
        return empresas;
    }else{
        document.getElementById("msgAlertaID").innerHTML = resposta['msg'];
    }
    } catch (error) {
        console.error(error);
    }
    
}
